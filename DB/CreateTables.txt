CREATE TABLE Customer (
  CustomerID Int NOT NULL AUTO INCREMENT PRIMARY KEY,
  Username Varchar(255) NOT NULL,
  Password Varchar(255) NOT NULL
);

CREATE TABLE Room (
  RoomID Int NOT NULL AUTO INCREMENT PRIMARY KEY,
  RoomName Varchar(255) NOT NULL,
  CustomerID Int NOT NULL FOREIGN KEY REFERENCES Customer(CustomerID),
  FloorNumber Int NOT NULL FOREIGN KEY REFERENCES Floor(FloorNumber),
  RoomIDNorth Int FOREIGN KEY REFERENCES Room(RoomID),
  DoorNorth Boolean,
  RoomIDEast Int FOREIGN KEY REFERENCES Room(RoomID),
  DoorEast Boolean,
  RoomIDSouth Int FOREIGN KEY REFERENCES Room(RoomID),
  DoorSouth Boolean,
  RoomIDWest Int FOREIGN KEY REFERENCES Room(RoomID),
  DoorWest Boolean
);

CREATE TABLE Floor (
  FloorID Int NOT NULL AUTO INCREMENT PRIMARY KEY,
  CustomerID Int NOT NULL FOREIGN KEY REFERENCES Customer(CustomerID),
  FloorNumber Int NOT NULL
);

CREATE TABLE SensorType (
  SensorTypeID Int NOT NULL PRIMARY KEY,
  SensorName Varchar(255) NOT NULL
);

CREATE TABLE Sensor (
  SensorID Int NOT NULL AUTO INCREMENT PRIMARY KEY,
  SensorTypeID Int NOT NULL FOREIGN KEY REFERENCES SensorType(SensorTypeID),
  RoomID Int NOT NULL FOREIGN KEY REFERENCES Room(RoomID)
);
